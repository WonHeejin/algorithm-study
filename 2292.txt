import java.util.Scanner;
public class Main{
    public static void main(String[] arg){
        Scanner sc= new Scanner(System.in);
        int n=sc.nextInt();
        int y=1;
        int x=1;
        while(n<=1000000000){
            if(n<=x){
                System.out.print(y);

        }
    }
}
